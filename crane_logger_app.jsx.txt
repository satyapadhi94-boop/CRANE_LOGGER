import React, { useState, useEffect, useMemo, useCallback } from 'react';
import { initializeApp } from 'firebase/app';
import { 
  getAuth, 
  signInAnonymously, 
  signInWithCustomToken, 
  onAuthStateChanged 
} from 'firebase/auth';
import { 
  getFirestore, 
  collection, 
  addDoc, 
  onSnapshot, 
  doc, 
  Timestamp,
  setDoc,
  query,
  where,
  getDocs,
  deleteDoc
} from 'firebase/firestore';
import { 
  Plus, 
  X, 
  Wrench, 
  TriangleAlert, 
  FileText, 
  Save, 
  Trash2, 
  Edit2,
  ChevronDown,
  ChevronUp,
  Package,
  Calendar,
  MapPin,
  Hash
} from 'lucide-react';

// --- Firebase Configuration ---
// These variables are injected by the environment.
const firebaseConfig = JSON.parse(typeof __firebase_config !== 'undefined' ? __firebase_config : '{}');
const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-crane-logger';
const initialAuthToken = typeof __initial_auth_token !== 'undefined' ? __initial_auth_token : null;

// --- Helper Components ---

// Simple Modal Component
const Modal = ({ children, onClose, title = "Data" }) => (
  <div 
    className="fixed inset-0 z-50 flex items-center justify-center bg-black bg-opacity-50 backdrop-blur-sm"
    onClick={onClose}
  >
    <div 
      className="bg-white rounded-xl shadow-2xl w-full max-w-2xl max-h-[90vh] overflow-y-auto"
      onClick={(e) => e.stopPropagation()}
    >
      <div className="sticky top-0 flex justify-between items-center p-4 border-b bg-gray-50 rounded-t-xl">
        <h2 className="text-xl font-semibold">{title}</h2>
        <button 
          onClick={onClose} 
          className="text-gray-500 hover:text-gray-800"
        >
          <X size={24} />
        </button>
      </div>
      <div className="p-6">
        {children}
      </div>
    </div>
  </div>
);

// Form Input Component
const Input = ({ label, id, value, onChange, placeholder, type = 'text' }) => (
  <div>
    <label htmlFor={id} className="block text-sm font-medium text-gray-700 mb-1">
      {label}
    </label>
    <input
      type={type}
      id={id}
      value={value}
      onChange={onChange}
      placeholder={placeholder}
      className="w-full px-3 py-2 border border-gray-300 rounded-md shadow-sm focus:outline-none focus:ring-blue-500 focus:border-blue-500"
    />
  </div>
);

// Textarea Component
const Textarea = ({ label, id, value, onChange, placeholder, rows = 4 }) => (
  <div>
    <label htmlFor={id} className="block text-sm font-medium text-gray-700 mb-1">
      {label}
    </label>
    <textarea
      id={id}
      value={value}
      onChange={onChange}
      placeholder={placeholder}
      rows={rows}
      className="w-full px-3 py-2 border border-gray-300 rounded-md shadow-sm focus:outline-none focus:ring-blue-500 focus:border-blue-500 resize-y"
    />
  </div>
);

// Select Component
const Select = ({ label, id, value, onChange, children }) => (
  <div>
    <label htmlFor={id} className="block text-sm font-medium text-gray-700 mb-1">
      {label}
    </label>
    <select
      id={id}
      value={value}
      onChange={onChange}
      className="w-full px-3 py-2 border border-gray-300 rounded-md shadow-sm focus:outline-none focus:ring-blue-500 focus:border-blue-500"
    >
      {children}
    </select>
  </div>
);

// Button Component
const Button = ({ onClick, children, variant = 'primary', className = '', disabled = false }) => {
  const baseStyle = "flex items-center justify-center gap-2 px-4 py-2 rounded-md font-semibold shadow-sm transition-all duration-150 focus:outline-none focus:ring-2 focus:ring-offset-2 disabled:opacity-50 disabled:cursor-not-allowed";
  const variants = {
    primary: 'bg-blue-600 text-white hover:bg-blue-700 focus:ring-blue-500',
    secondary: 'bg-gray-200 text-gray-800 hover:bg-gray-300 focus:ring-gray-400',
    danger: 'bg-red-600 text-white hover:bg-red-700 focus:ring-red-500',
    ghost: 'bg-transparent text-gray-600 hover:bg-gray-100 focus:ring-gray-200',
  };
  return (
    <button onClick={onClick} disabled={disabled} className={`${baseStyle} ${variants[variant]} ${className}`}>
      {children}
    </button>
  );
};

// --- Main Application ---

const App = () => {
  // --- State ---
  // Firebase & Auth
  const [db, setDb] = useState(null);
  const [auth, setAuth] = useState(null);
  const [userId, setUserId] = useState(null);
  const [isAuthReady, setIsAuthReady] = useState(false);

  // App Data
  const [cranes, setCranes] = useState([]);
  const [selectedCrane, setSelectedCrane] = useState(null);
  const [events, setEvents] = useState([]);
  const [materials, setMaterials] = useState([]); // New state for materials
  
  // UI State
  const [isLoading, setIsLoading] = useState(true);
  const [error, setError] = useState(null);
  const [isCraneModalOpen, setIsCraneModalOpen] = useState(false);
  const [isEventModalOpen, setIsEventModalOpen] = useState(false);
  const [isMaterialModalOpen, setIsMaterialModalOpen] = useState(false); // New modal state
  
  const [isEditingCrane, setIsEditingCrane] = useState(false);
  const [isDeleting, setIsDeleting] = useState(false);
  const [searchTerm, setSearchTerm] = useState('');
  
  const [isEditingEvent, setIsEditingEvent] = useState(false);
  const [editingEventId, setEditingEventId] = useState(null);
  
  const [isEditingMaterial, setIsEditingMaterial] = useState(false); // New edit state
  const [editingMaterialId, setEditingMaterialId] = useState(null);

  const [isEventHistoryCollapsed, setIsEventHistoryCollapsed] = useState(false);
  const [showMaintenance, setShowMaintenance] = useState(true);
  const [showBreakdowns, setShowBreakdowns] = useState(true);
  
  // Tab State: 'events' or 'materials'
  const [activeTab, setActiveTab] = useState('events');

  // --- Utility Functions ---
  function getInitialCraneForm() {
    return {
      name: '',
      craneType: '',
      motionType: '',
      rating: '',
      make: '',
      motorDriveData: '',
      plcData: '',
      maintenanceSchedule: '',
      lubricationSchedule: '',
      otherNotes: '',
    };
  }

  // Helper functions to format Date object for form inputs
  const toYYYYMMDD = (date) => {
    if (!date) return '';
    return date.toISOString().split('T')[0];
  }
  
  const toHHMM = (date) => {
    if (!date) return '';
    return date.toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit', hour12: false });
  }

  function getInitialEventForm() {
    return {
      type: 'breakdown',
      date: toYYYYMMDD(new Date()),
      time: toHHMM(new Date()),
      shift: '',
      description: '',
      resolution: '',
    };
  }

  function getInitialMaterialForm() {
    return {
      name: '',
      spareNo: '',
      location: '',
      lastChange: toYYYYMMDD(new Date()),
    };
  }

  // Form State
  const [craneForm, setCraneForm] = useState(getInitialCraneForm());
  const [eventForm, setEventForm] = useState(getInitialEventForm());
  const [materialForm, setMaterialForm] = useState(getInitialMaterialForm());

  // --- Firebase Initialization ---
  useEffect(() => {
    try {
      const app = initializeApp(firebaseConfig);
      const authInstance = getAuth(app);
      const dbInstance = getFirestore(app);
      
      setAuth(authInstance);
      setDb(dbInstance);

      onAuthStateChanged(authInstance, async (user) => {
        if (user) {
          setUserId(user.uid);
          setIsAuthReady(true);
        } else {
          try {
            if (initialAuthToken) {
              await signInWithCustomToken(authInstance, initialAuthToken);
            } else {
              await signInAnonymously(authInstance);
            }
          } catch (authError) {
            console.error("Authentication error:", authError);
            setError("Failed to authenticate. Please refresh.");
          }
        }
      });
    } catch (e) {
      console.error("Firebase init error:", e);
      setError("Failed to initialize app. Check Firebase config.");
    }
  }, []);

  // --- Data Paths ---
  const cranesColPath = useMemo(() => 
    isAuthReady ? `artifacts/${appId}/users/${userId}/cranes` : null
  , [isAuthReady, userId]);
  
  const eventsColPath = useMemo(() =>
    (isAuthReady && selectedCrane) ? `artifacts/${appId}/users/${userId}/cranes/${selectedCrane.id}/events` : null
  , [isAuthReady, userId, selectedCrane]);

  const materialsColPath = useMemo(() =>
    (isAuthReady && selectedCrane) ? `artifacts/${appId}/users/${userId}/cranes/${selectedCrane.id}/materials` : null
  , [isAuthReady, userId, selectedCrane]);

  // --- Data Fetching (Cranes) ---
  useEffect(() => {
    if (!cranesColPath) return;

    setIsLoading(true);
    const q = query(collection(db, cranesColPath));
    const unsubscribe = onSnapshot(q, (snapshot) => {
      const cranesData = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
      // Client-side sorting (orderBy is disallowed)
      cranesData.sort((a, b) => a.name.localeCompare(b.name));
      setCranes(cranesData);
      setIsLoading(false);
    }, (err) => {
      console.error("Error fetching cranes:", err);
      setError("Failed to fetch cranes.");
      setIsLoading(false);
    });

    return () => unsubscribe();
  }, [cranesColPath, db]);

  // --- Data Fetching (Events) ---
  useEffect(() => {
    if (!eventsColPath) {
      setEvents([]);
      return;
    }

    const q = query(collection(db, eventsColPath));
    const unsubscribe = onSnapshot(q, (snapshot) => {
      const eventsData = snapshot.docs.map(doc => {
        const data = doc.data();
        return {
          id: doc.id,
          ...data,
          // Convert Firestore Timestamp to JS Date
          timestamp: data.timestamp?.toDate()
        };
      });
      
      // Client-side sorting (orderBy is disallowed)
      eventsData.sort((a, b) => b.timestamp - a.timestamp); // Newest first
      
      setEvents(eventsData);
    }, (err) => {
      console.error("Error fetching events:", err);
      setError("Failed to fetch events for this crane.");
    });

    return () => unsubscribe();
  }, [eventsColPath, db]);

  // --- Data Fetching (Materials) ---
  useEffect(() => {
    if (!materialsColPath) {
      setMaterials([]);
      return;
    }

    const q = query(collection(db, materialsColPath));
    const unsubscribe = onSnapshot(q, (snapshot) => {
      const materialsData = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
      // Sort by name
      materialsData.sort((a, b) => a.name.localeCompare(b.name));
      setMaterials(materialsData);
    }, (err) => {
      console.error("Error fetching materials:", err);
      setError("Failed to fetch material list.");
    });

    return () => unsubscribe();
  }, [materialsColPath, db]);

  // --- Event Handlers ---

  const handleSelectCrane = (crane) => {
    setSelectedCrane(crane);
    setActiveTab('events'); // Default to events tab
  };

  const handleOpenCraneModal = (craneToEdit = null) => {
    if (craneToEdit) {
      setIsEditingCrane(true);
      setCraneForm({ ...getInitialCraneForm(), ...craneToEdit });
    } else {
      setIsEditingCrane(false);
      setCraneForm(getInitialCraneForm());
    }
    setIsCraneModalOpen(true);
  };

  const handleCloseCraneModal = () => {
    setIsCraneModalOpen(false);
    setIsEditingCrane(false);
    setCraneForm(getInitialCraneForm());
    setError(null);
  };

  const handleSaveCrane = async () => {
    if (!craneForm.name || !craneForm.craneType) {
      setError("Crane Name and Type are required.");
      return;
    }
    if (!cranesColPath) {
      setError("Database not ready.");
      return;
    }

    try {
      if (isEditingCrane && craneForm.id) {
        // Update existing crane
        const craneRef = doc(db, cranesColPath, craneForm.id);
        await setDoc(craneRef, craneForm, { merge: true });
      } else {
        // Add new crane
        await addDoc(collection(db, cranesColPath), craneForm);
      }
      handleCloseCraneModal();
    } catch (e) {
      console.error("Error saving crane:", e);
      setError("Failed to save crane data.");
    }
  };
  
  const handleDeleteCrane = async (craneId) => {
    if (!window.confirm("Are you sure you want to delete this crane and all its log history? This cannot be undone.")) {
      return;
    }
    if (!cranesColPath) {
      setError("Database not ready.");
      return;
    }
    
    setIsDeleting(true);
    try {
      // 1. Delete Events
      const eventsPath = `artifacts/${appId}/users/${userId}/cranes/${craneId}/events`;
      const eventsQuery = query(collection(db, eventsPath));
      const eventsSnapshot = await getDocs(eventsQuery);
      const deletePromises = eventsSnapshot.docs.map(d => deleteDoc(d.ref));
      
      // 2. Delete Materials
      const materialsPath = `artifacts/${appId}/users/${userId}/cranes/${craneId}/materials`;
      const materialsQuery = query(collection(db, materialsPath));
      const materialsSnapshot = await getDocs(materialsQuery);
      deletePromises.push(...materialsSnapshot.docs.map(d => deleteDoc(d.ref)));

      await Promise.all(deletePromises);

      // 3. Delete Crane
      const craneRef = doc(db, cranesColPath, craneId);
      await deleteDoc(craneRef);
      
      setSelectedCrane(null);
      handleCloseCraneModal();

    } catch (e) {
      console.error("Error deleting crane:", e);
      setError("Failed to delete crane.");
    } finally {
      setIsDeleting(false);
    }
  };

  // --- Event Log Handlers ---

  const handleOpenEventModal = (eventToEdit = null) => {
    setError(null);
    if (eventToEdit) {
      setIsEditingEvent(true);
      setEditingEventId(eventToEdit.id);
      setEventForm({
        type: eventToEdit.type,
        date: toYYYYMMDD(eventToEdit.timestamp),
        time: toHHMM(eventToEdit.timestamp),
        shift: eventToEdit.shift,
        description: eventToEdit.description,
        resolution: eventToEdit.resolution || '',
      });
    } else {
      setIsEditingEvent(false);
      setEditingEventId(null);
      setEventForm(getInitialEventForm());
    }
    setIsEventModalOpen(true);
  };

  const handleCloseEventModal = () => {
    setIsEventModalOpen(false);
    setEventForm(getInitialEventForm());
    setError(null);
    setIsEditingEvent(false);
    setEditingEventId(null);
  };

  const handleSaveEvent = async () => {
    if (!eventForm.description || !eventForm.shift) {
      setError("Description and Shift are required.");
      return;
    }
    if (!eventsColPath) {
      setError("No crane selected or database not ready.");
      return;
    }
    
    const dateTimeString = `${eventForm.date}T${eventForm.time}:00`;
    const eventTimestamp = new Date(dateTimeString);

    const eventData = {
      type: eventForm.type,
      timestamp: Timestamp.fromDate(eventTimestamp),
      shift: eventForm.shift,
      description: eventForm.description,
      resolution: eventForm.resolution,
    };

    try {
      if (isEditingEvent && editingEventId) {
        const eventRef = doc(db, eventsColPath, editingEventId);
        await setDoc(eventRef, eventData, { merge: true });
      } else {
        await addDoc(collection(db, eventsColPath), eventData);
      }
      handleCloseEventModal();
    } catch (e) {
      console.error("Error saving event:", e);
      setError("Failed to save event log.");
    }
  };

  const handleDeleteEvent = async () => {
    if (!editingEventId || !eventsColPath) return;
    if (!window.confirm("Delete this event log?")) return;
    
    try {
      const eventRef = doc(db, eventsColPath, editingEventId);
      await deleteDoc(eventRef);
      handleCloseEventModal();
    } catch (e) {
      console.error("Error deleting event:", e);
      setError("Failed to delete event.");
    }
  };

  // --- Material List Handlers ---

  const handleOpenMaterialModal = (materialToEdit = null) => {
    setError(null);
    if (materialToEdit) {
      setIsEditingMaterial(true);
      setEditingMaterialId(materialToEdit.id);
      setMaterialForm({
        name: materialToEdit.name,
        spareNo: materialToEdit.spareNo || '',
        location: materialToEdit.location || '',
        lastChange: materialToEdit.lastChange || toYYYYMMDD(new Date()),
      });
    } else {
      setIsEditingMaterial(false);
      setEditingMaterialId(null);
      setMaterialForm(getInitialMaterialForm());
    }
    setIsMaterialModalOpen(true);
  };

  const handleCloseMaterialModal = () => {
    setIsMaterialModalOpen(false);
    setIsEditingMaterial(false);
    setEditingMaterialId(null);
    setMaterialForm(getInitialMaterialForm());
    setError(null);
  };

  const handleSaveMaterial = async () => {
    if (!materialForm.name) {
      setError("Material Name is required.");
      return;
    }
    if (!materialsColPath) {
      setError("Database error.");
      return;
    }

    try {
      if (isEditingMaterial && editingMaterialId) {
        const matRef = doc(db, materialsColPath, editingMaterialId);
        await setDoc(matRef, materialForm, { merge: true });
      } else {
        await addDoc(collection(db, materialsColPath), materialForm);
      }
      handleCloseMaterialModal();
    } catch (e) {
      console.error("Error saving material:", e);
      setError("Failed to save material.");
    }
  };

  const handleDeleteMaterial = async () => {
    if (!editingMaterialId || !materialsColPath) return;
    if (!window.confirm("Delete this material?")) return;

    try {
      const matRef = doc(db, materialsColPath, editingMaterialId);
      await deleteDoc(matRef);
      handleCloseMaterialModal();
    } catch (e) {
      console.error("Error deleting material:", e);
      setError("Failed to delete material.");
    }
  };


  // --- Form Updaters ---
  const handleCraneFormChange = (e) => {
    const { id, value } = e.target;
    setCraneForm(prev => ({ ...prev, [id]: value }));
  };

  const handleEventFormChange = (e) => {
    const { id, value } = e.target;
    setEventForm(prev => ({ ...prev, [id]: value }));
  };

  const handleMaterialFormChange = (e) => {
    const { id, value } = e.target;
    setMaterialForm(prev => ({ ...prev, [id]: value }));
  };
  
  // --- Filtering ---
  const filteredCranes = useMemo(() => {
    return cranes.filter(crane => 
      crane.name.toLowerCase().includes(searchTerm.toLowerCase()) ||
      crane.craneType.toLowerCase().includes(searchTerm.toLowerCase()) ||
      crane.make.toLowerCase().includes(searchTerm.toLowerCase())
    );
  }, [cranes, searchTerm]);

  // --- Filtering Events ---
  const handleToggleAllEvents = (checked) => {
    setShowMaintenance(checked);
    setShowBreakdowns(checked);
  };

  const allEventsChecked = showMaintenance && showBreakdowns;

  const filteredEvents = useMemo(() => {
    return events.filter(event => {
      if (event.type === 'maintenance' && !showMaintenance) {
        return false;
      }
      if (event.type === 'breakdown' && !showBreakdowns) {
        return false;
      }
      return true;
    });
  }, [events, showMaintenance, showBreakdowns]);

  // --- Render ---

  if (!isAuthReady) {
    return (
      <div className="flex items-center justify-center min-h-screen bg-gray-100">
        <div className="text-xl font-semibold text-gray-700">Loading Logger...</div>
      </div>
    );
  }

  return (
    <div className="flex h-screen bg-gray-100 font-inter">
      {/* --- Left Sidebar: Crane List --- */}
      <aside className="w-80 h-full bg-white shadow-md flex flex-col">
        <div className="p-4 border-b">
          <h1 className="text-2xl font-bold text-blue-800">Crane Logger</h1>
          <p className="text-sm text-gray-500">Your maintenance dashboard</p>
        </div>
        
        <div className="p-4 border-b">
           <Button onClick={() => handleOpenCraneModal()} className="w-full">
            <Plus size={18} /> Add New Crane
          </Button>
          <input
            type="text"
            placeholder="Search cranes..."
            value={searchTerm}
            onChange={(e) => setSearchTerm(e.target.value)}
            className="w-full px-3 py-2 border border-gray-300 rounded-md shadow-sm mt-4 focus:outline-none focus:ring-blue-500 focus:border-blue-500"
          />
        </div>

        <nav className="flex-1 overflow-y-auto">
          {isLoading ? (
            <div className="p-4 text-center text-gray-500">Loading cranes...</div>
          ) : filteredCranes.length === 0 ? (
             <div className="p-4 text-center text-gray-500">No cranes found.</div>
          ) : (
            <ul>
              {filteredCranes.map(crane => (
                <li key={crane.id}>
                  <button
                    onClick={() => handleSelectCrane(crane)}
                    className={`w-full text-left px-4 py-3 ${selectedCrane?.id === crane.id ? 'bg-blue-100 text-blue-700 font-semibold' : 'text-gray-700 hover:bg-gray-50'}`}
                  >
                    {crane.name}
                    <span className="block text-xs text-gray-500">{crane.craneType}</span>
                  </button>
                </li>
              ))}
            </ul>
          )}
        </nav>
        
        <div className="p-2 text-center text-xs text-gray-400 border-t">
          User ID: <span className="font-mono">{userId.substring(0, 10)}...</span>
        </div>
      </aside>

      {/* --- Main Content: Details & Logs --- */}
      <main className="flex-1 flex flex-col h-full overflow-y-auto">
        {!selectedCrane ? (
          <div className="flex-1 flex items-center justify-center text-gray-500 text-lg">
            Select a crane to view details or add a new crane
          </div>
        ) : (
          <>
            {/* Crane Details Header */}
            <div className="sticky top-0 bg-white p-4 border-b shadow-sm z-10">
              <div className="flex justify-between items-center">
                <div>
                  <h2 className="text-3xl font-bold text-gray-800">{selectedCrane.name}</h2>
                  <p className="text-md text-gray-600">
                    {selectedCrane.make} - {selectedCrane.craneType} - {selectedCrane.rating}
                  </p>
                </div>
                <div className="flex gap-2">
                  <Button onClick={() => handleOpenCraneModal(selectedCrane)} variant="secondary">
                    <Edit2 size={18} /> Edit Info
                  </Button>
                </div>
              </div>
            </div>

            {/* Content Area */}
            <div className="flex-1 p-6 grid grid-cols-12 gap-6">
              
              {/* Left Column: Crane Info */}
              <div className="col-span-12 md:col-span-4 lg:col-span-3 bg-white p-6 rounded-xl shadow-lg h-fit">
                <h3 className="text-xl font-semibold mb-4 border-b pb-2">Crane Details</h3>
                <div className="space-y-3 text-sm">
                  <InfoItem label="Motion Type" value={selectedCrane.motionType} />
                  <InfoItem label="Maintenance" value={selectedCrane.maintenanceSchedule} />
                  <InfoItem label="Lubrication" value={selectedCrane.lubricationSchedule} />
                  <InfoItem label="Motor Drive" value={selectedCrane.motorDriveData} isMultiline={true} />
                  <InfoItem label="PLC Data" value={selectedCrane.plcData} isMultiline={true} />
                  <InfoItem label="Other Notes" value={selectedCrane.otherNotes} isMultiline={true} />
                </div>
              </div>

              {/* Right Column: Tabs (Events & Materials) */}
              <div className="col-span-12 md:col-span-8 lg:col-span-9 flex flex-col gap-4">
                
                {/* Tab Navigation */}
                <div className="flex border-b border-gray-300">
                  <button
                    onClick={() => setActiveTab('events')}
                    className={`px-6 py-3 font-medium text-sm focus:outline-none transition-colors duration-200 ${
                      activeTab === 'events'
                        ? 'border-b-2 border-blue-600 text-blue-600'
                        : 'text-gray-500 hover:text-gray-700'
                    }`}
                  >
                    Event History
                  </button>
                  <button
                    onClick={() => setActiveTab('materials')}
                    className={`px-6 py-3 font-medium text-sm focus:outline-none transition-colors duration-200 ${
                      activeTab === 'materials'
                        ? 'border-b-2 border-blue-600 text-blue-600'
                        : 'text-gray-500 hover:text-gray-700'
                    }`}
                  >
                    Material List (Spare Parts)
                  </button>
                </div>

                {/* Tab Content: Events */}
                {activeTab === 'events' && (
                  <div className="space-y-4">
                    <div className="flex justify-between items-center">
                      <h3 className="text-2xl font-semibold text-gray-800">Event History</h3>
                      <div className="flex gap-2">
                         <Button onClick={handleOpenEventModal} variant="primary">
                          <Plus size={18} /> Log Event
                        </Button>
                        <Button 
                          onClick={() => setIsEventHistoryCollapsed(prev => !prev)} 
                          variant="secondary"
                          className="px-2 py-1 text-sm"
                        >
                          {isEventHistoryCollapsed ? <ChevronDown size={18} /> : <ChevronUp size={18} />}
                          {isEventHistoryCollapsed ? 'Show' : 'Hide'}
                        </Button>
                      </div>
                    </div>
                    
                    {!isEventHistoryCollapsed && (
                      <>
                         {/* Filter Toggles */}
                        <div className="flex flex-wrap gap-4 items-center p-4 bg-white rounded-xl shadow-sm border border-gray-100">
                          <span className="font-semibold text-gray-700 mr-2">Filters:</span>
                          
                          {/* ALL Toggle */}
                           <div className="flex items-center gap-2 pr-4 border-r border-gray-200">
                            <input
                              type="checkbox"
                              id="showAll"
                              checked={allEventsChecked}
                              onChange={(e) => handleToggleAllEvents(e.target.checked)}
                              className="h-4 w-4 rounded text-gray-800 focus:ring-gray-500 border-gray-300"
                            />
                            <label htmlFor="showAll" className="text-sm text-gray-800 font-bold">
                              All Events
                            </label>
                          </div>

                          <div className="flex items-center gap-2">
                            <input
                              type="checkbox"
                              id="showBreakdowns"
                              checked={showBreakdowns}
                              onChange={(e) => setShowBreakdowns(e.target.checked)}
                              className="h-4 w-4 rounded text-red-600 focus:ring-red-500 border-gray-300"
                            />
                            <label htmlFor="showBreakdowns" className="text-sm text-red-700 font-medium">
                              Breakdowns
                            </label>
                          </div>
                          
                          <div className="flex items-center gap-2">
                            <input
                              type="checkbox"
                              id="showMaintenance"
                              checked={showMaintenance}
                              onChange={(e) => setShowMaintenance(e.target.checked)}
                              className="h-4 w-4 rounded text-blue-600 focus:ring-blue-500 border-gray-300"
                            />
                            <label htmlFor="showMaintenance" className="text-sm text-blue-700 font-medium">
                              Maintenance
                            </label>
                          </div>
                        </div>

                        {/* List */}
                        {events.length === 0 ? (
                          <div className="bg-white p-6 rounded-xl shadow-lg text-center text-gray-500">
                            No events logged for this crane yet.
                          </div>
                        ) : filteredEvents.length === 0 ? (
                           <div className="bg-white p-6 rounded-xl shadow-lg text-center text-gray-500">
                            No events match the current filters.
                          </div>
                        ) : (
                          <div className="space-y-4">
                            {filteredEvents.map(event => (
                              <EventCard 
                                key={event.id} 
                                event={event} 
                                onEdit={() => handleOpenEventModal(event)} 
                              />
                            ))}
                          </div>
                        )}
                      </>
                    )}
                  </div>
                )}

                {/* Tab Content: Materials */}
                {activeTab === 'materials' && (
                  <div className="space-y-4">
                    <div className="flex justify-between items-center">
                      <h3 className="text-2xl font-semibold text-gray-800">Material List</h3>
                      <Button onClick={() => handleOpenMaterialModal()} variant="primary">
                        <Plus size={18} /> Add Material
                      </Button>
                    </div>

                    <div className="bg-white rounded-xl shadow-lg overflow-hidden">
                      <div className="overflow-x-auto">
                        <table className="min-w-full divide-y divide-gray-200">
                          <thead className="bg-gray-50">
                            <tr>
                              <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">Material Name</th>
                              <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">Spare No</th>
                              <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">Location</th>
                              <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">Last Change</th>
                              <th className="px-6 py-3 text-right text-xs font-medium text-gray-500 uppercase tracking-wider">Actions</th>
                            </tr>
                          </thead>
                          <tbody className="bg-white divide-y divide-gray-200">
                            {materials.length === 0 ? (
                              <tr>
                                <td colSpan="5" className="px-6 py-4 text-center text-sm text-gray-500">
                                  No materials or spare parts listed.
                                </td>
                              </tr>
                            ) : (
                              materials.map((mat) => (
                                <tr key={mat.id} className="hover:bg-gray-50">
                                  <td className="px-6 py-4 whitespace-nowrap text-sm font-medium text-gray-900 flex items-center gap-2">
                                    <Package size={16} className="text-gray-400" />
                                    {mat.name}
                                  </td>
                                  <td className="px-6 py-4 whitespace-nowrap text-sm text-gray-500">{mat.spareNo || '-'}</td>
                                  <td className="px-6 py-4 whitespace-nowrap text-sm text-gray-500">{mat.location || '-'}</td>
                                  <td className="px-6 py-4 whitespace-nowrap text-sm text-gray-500">{mat.lastChange || '-'}</td>
                                  <td className="px-6 py-4 whitespace-nowrap text-right text-sm font-medium">
                                    <button 
                                      onClick={() => handleOpenMaterialModal(mat)}
                                      className="text-indigo-600 hover:text-indigo-900 flex items-center justify-end w-full"
                                    >
                                      <Edit2 size={16} />
                                    </button>
                                  </td>
                                </tr>
                              ))
                            )}
                          </tbody>
                        </table>
                      </div>
                    </div>
                  </div>
                )}
              </div>
            </div>
          </>
        )}
      </main>

      {/* --- Modals --- */}
      
      {/* Add/Edit Crane Modal */}
      {isCraneModalOpen && (
        <Modal 
          onClose={handleCloseCraneModal}
          title={isEditingCrane ? 'Edit Crane' : 'Add New Crane'}
        >
          <div className="space-y-4">
            {error && <div className="text-red-600 bg-red-100 p-3 rounded-md">{error}</div>}
            
            <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
              <Input label="Crane Name" id="name" value={craneForm.name} onChange={handleCraneFormChange} placeholder="e.g., EOT Crane 01" />
              <Input label="Type of Crane" id="craneType" value={craneForm.craneType} onChange={handleCraneFormChange} placeholder="e.g., Double Girder" />
              <Input label="Type of Motion" id="motionType" value={craneForm.motionType} onChange={handleCraneFormChange} placeholder="e.g., All Electric" />
              <Input label="Rating" id="rating" value={craneForm.rating} onChange={handleCraneFormChange} placeholder="e.g., 10 Ton" />
              <Input label="Make" id="make" value={craneForm.make} onChange={handleCraneFormChange} placeholder="e.g., Konecranes" />
              <Input label="Maintenance Schedule" id="maintenanceSchedule" value={craneForm.maintenanceSchedule} onChange={handleCraneFormChange} placeholder="e.g., Quarterly" />
              <Input label="Lubrication Schedule" id="lubricationSchedule" value={craneForm.lubricationSchedule} onChange={handleCraneFormChange} placeholder="e.g., Monthly" />
            </div>
            
            <Textarea label="Motor Drive Data" id="motorDriveData" value={craneForm.motorDriveData} onChange={handleCraneFormChange} placeholder="Enter all motor and drive details..." />
            <Textarea label="PLC Data" id="plcData" value={craneForm.plcData} onChange={handleCraneFormChange} placeholder="Enter PLC model, program version, etc..." />
            <Textarea label="Other Notes" id="otherNotes" value={craneForm.otherNotes} onChange={handleCraneFormChange} placeholder="Any other relevant information..." />
            
            <div className="flex justify-between items-center pt-4 border-t mt-6">
              <div>
                {isEditingCrane && (
                  <Button 
                    onClick={() => handleDeleteCrane(craneForm.id)} 
                    variant="danger"
                    disabled={isDeleting}
                  >
                    <Trash2 size={18} /> 
                    {isDeleting ? 'Deleting...' : 'Delete Crane'}
                  </Button>
                )}
              </div>
              <div className="flex gap-2">
                <Button onClick={handleCloseCraneModal} variant="secondary">Cancel</Button>
                <Button onClick={handleSaveCrane} variant="primary">
                  <Save size={18} /> Save Crane
                </Button>
              </div>
            </div>
          </div>
        </Modal>
      )}

      {/* Log Event Modal */}
      {isEventModalOpen && (
        <Modal 
          onClose={handleCloseEventModal}
          title={isEditingEvent ? 'Edit Event Log' : 'Log New Event'}
        >
          <div className="space-y-4">
            {error && <div className="text-red-600 bg-red-100 p-3 rounded-md">{error}</div>}
            
            <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
              <Select label="Event Type" id="type" value={eventForm.type} onChange={handleEventFormChange}>
                <option value="breakdown">Breakdown</option>
                <option value="maintenance">Maintenance</option>
              </Select>
              <Input label="Date" id="date" value={eventForm.date} onChange={handleEventFormChange} type="date" />
              <Input label="Time" id="time" value={eventForm.time} onChange={handleEventFormChange} type="time" />
            </div>

            <Select label="Shift" id="shift" value={eventForm.shift} onChange={handleEventFormChange}>
              <option value="">Select Shift</option>
              <option value="A">Shift A</option>
              <option value="B">Shift B</option>
              <option value="C">Shift C</option>
              <option value="General">General</option>
            </Select>
            
            <Textarea label="Description" id="description" value={eventForm.description} onChange={handleEventFormChange} placeholder={eventForm.type === 'breakdown' ? 'Describe the breakdown...' : 'Describe the maintenance performed...'} />
            
            {eventForm.type === 'breakdown' && (
              <Textarea label="Resolution" id="resolution" value={eventForm.resolution} onChange={handleEventFormChange} placeholder="How was the breakdown solved?" />
            )}
            
            <div className="flex justify-between items-center pt-4 border-t mt-6">
              <div>
                {isEditingEvent && (
                  <Button
                    onClick={handleDeleteEvent}
                    variant="danger"
                  >
                    <Trash2 size={18} /> Delete Event
                  </Button>
                )}
              </div>
              <div className="flex gap-2">
                <Button onClick={handleCloseEventModal} variant="secondary">Cancel</Button>
                <Button onClick={handleSaveEvent} variant="primary">
                  <Save size={18} /> {isEditingEvent ? 'Update Event' : 'Log Event'}
                </Button>
              </div>
            </div>
          </div>
        </Modal>
      )}

       {/* Material Modal */}
       {isMaterialModalOpen && (
        <Modal 
          onClose={handleCloseMaterialModal}
          title={isEditingMaterial ? 'Edit Material' : 'Add New Material'}
        >
          <div className="space-y-4">
            {error && <div className="text-red-600 bg-red-100 p-3 rounded-md">{error}</div>}
            
            <Input label="Material Name" id="name" value={materialForm.name} onChange={handleMaterialFormChange} placeholder="e.g., Brake Pad, Wire Rope" />
            
            <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
              <Input label="Spare No" id="spareNo" value={materialForm.spareNo} onChange={handleMaterialFormChange} placeholder="e.g., SP-10234" />
              <Input label="Location" id="location" value={materialForm.location} onChange={handleMaterialFormChange} placeholder="e.g., Rack A-4" />
            </div>

            <Input label="Last Change Date" id="lastChange" value={materialForm.lastChange} onChange={handleMaterialFormChange} type="date" />
            
            <div className="flex justify-between items-center pt-4 border-t mt-6">
              <div>
                {isEditingMaterial && (
                  <Button
                    onClick={handleDeleteMaterial}
                    variant="danger"
                  >
                    <Trash2 size={18} /> Delete Material
                  </Button>
                )}
              </div>
              <div className="flex gap-2">
                <Button onClick={handleCloseMaterialModal} variant="secondary">Cancel</Button>
                <Button onClick={handleSaveMaterial} variant="primary">
                  <Save size={18} /> {isEditingMaterial ? 'Update Material' : 'Add Material'}
                </Button>
              </div>
            </div>
          </div>
        </Modal>
      )}

    </div>
  );
};

// --- Helper Components for Main App ---

const InfoItem = ({ label, value, isMultiline = false }) => {
  if (!value) return null;
  return (
    <div>
      <strong className="text-gray-500">{label}:</strong>
      {isMultiline ? (
        <p className="whitespace-pre-wrap bg-gray-50 p-2 rounded-md mt-1 font-mono text-gray-800">{value}</p>
      ) : (
        <span className="text-gray-800 ml-2">{value}</span>
      )}
    </div>
  );
};

const EventCard = ({ event, onEdit }) => {
  const isBreakdown = event.type === 'breakdown';
  const eventDate = event.timestamp ? event.timestamp.toLocaleDateString() : 'No date';
  const eventTime = event.timestamp ? event.timestamp.toLocaleTimeString() : 'No time';
  
  return (
    <div className="bg-white rounded-xl shadow-lg transition-all hover:shadow-xl border border-gray-100">
      <div className={`flex items-center justify-between p-4 rounded-t-xl ${isBreakdown ? 'bg-red-50 text-red-800 border-b border-red-100' : 'bg-blue-50 text-blue-800 border-b border-blue-100'}`}>
        <div className="flex items-center gap-3">
          {isBreakdown ? <TriangleAlert size={24} className="text-red-600" /> : <Wrench size={24} className="text-blue-600" />}
          <h4 className="text-xl font-semibold">
            {isBreakdown ? 'Breakdown Reported' : 'Maintenance Logged'}
          </h4>
        </div>
        <Button onClick={onEdit} variant="secondary" className="px-2 py-1 text-sm bg-white hover:bg-white/80 shadow-sm border">
          <Edit2 size={16} /> Edit
        </Button>
      </div>
      <div className="p-6 grid grid-cols-3 gap-4">
        <InfoItem label="Date" value={eventDate} />
        <InfoItem label="Time" value={eventTime} />
        <InfoItem label="Shift" value={event.shift} />
        
        <div className="col-span-3">
          <InfoItem label="Description" value={event.description} isMultiline={true} />
        </div>
        
        {isBreakdown && (
          <div className="col-span-3">
            <InfoItem label="Resolution" value={event.resolution} isMultiline={true} />
          </div>
        )}
      </div>
    </div>
  );
};

export default App;
